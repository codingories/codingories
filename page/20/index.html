day2_Easy_07_Reverse_Integer

Posted on 2019-05-27 Edited on 2019-05-29 Views: Valine:

Given a 32-bit signed integer, reverse digits of an integer.

Example 1:

Input: 123 Output: 321 Example 2:

Input: -123 Output: -321 Example 3:

Input: 120 Output: 21 Note: Assume we are dealing with an environment which could only store integers within the 32-bit signed integer range: [−231, 231 − 1]. For the purpose of this problem, assume that your function returns 0 when the reversed integer overflows.

# 第一种方法，利用数字转字符串做
class Solution:
    def reverse(self, x: int) -> int:
        if  0 < x and x <  2**31 - 1:
            x = str(x)
            # print(x)
            x = x[::-1]
            x = int(x)
            if x > 2**31 - 1:
                return 0
            else:
                return x
        elif -2**31 - 1 < x and x < 0:
            x = x * (-1)
            x = str(x)
            x = x[::-1]
            x = '-'+x
            # print(x)
            x = int(x)
            if x < -2**31 - 1:
                return(0)
            else:
                return x
        else:
            return 0

# 第二种方法利用取余做
class Solution:
    def reverse(self, x: int) -> int:
        num = 0
        # 返回一个绝对值
        a = abs(x)
        while(a != 0):
            # 123
            # a = 123
            # num = 0
            # 第一次迭代
            # a = 12
            # num = 3
            # 第二次迭代
            # a = 1
            # num = 32
            # 第三次迭代
            # a = 0
            # num = 321
            temp = a % 10
            num = num * 10 + temp
            a = int(a/10)
        if x > 0 and num < 2147483647:
            return num
        elif x < 0 and num <= 2147483647:
            return -num
        else:
            return 0
